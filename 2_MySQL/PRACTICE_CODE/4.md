	student_id INT PRIMARY KEY AUTO_INCREMENT,

);

CREATE TABLE if not exists Courses(
	course_id BIGINT PRIMARY KEY AUTO_INCREMENT,
	course_name VARCHAR(100) NOT NULL, 
	credits INT check (credits>0)
);
