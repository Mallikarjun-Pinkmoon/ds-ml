
```sql

CREATE TABLE employees (
    id INT AUTO_INCREMENT PRIMARY KEY,
    full_name VARCHAR(150) NOT NULL,
    department_id INT NOT NULL,
    job_title VARCHAR(100),
    salary DECIMAL(10,2),
    hire_date DATE,
    email VARCHAR(150) UNIQUE
);
```
