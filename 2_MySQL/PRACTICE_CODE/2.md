-- DATA TYPES
USE School;

DROP TABLE Students;

CREATE TABLE IF NOT EXISTS Students (
	id BIGINT PRIMARY KEY AUTO_INCREMENT,
    class SMALLINT CHECK ((class > 0) AND (class < 13)) NOT NULL,
    fullName VARCHAR(50) NOT NULL,
    section ENUM("A", "B", "C") NOT NULL,
    comp_skills SET("PY", "JAVA", "C", "SQL"),
    bio TEXT,
    gender ENUM("M", "F") NOT NULL,
    ph_no VARCHAR(10) NOT NULL UNIQUE
);

DESC Students;
